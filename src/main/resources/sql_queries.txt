use payroll;

create table employee_table(emp_id integer primary key , emp_name varchar(30), email varchar(20), password varchar(15), department varchar(20), designation varchar(20),mobile_number varchar(15),address varchar(100));

create table deduction_table(emp_id integer, type_of_deduction varchar(30), deduction_amount integer,foreign key(emp_id) references employee_table(emp_id));

create table attandance_table(emp_id integer,no_of_leaves integer,leaves_taken integer,month varchar(15),year integer,foreign key(emp_id) references employee_table(emp_id));

create table payroll_table(pay_id integer primary key,emp_id integer,pf_no integer, salary integer,datefrom varchar(25),dateto varchar(25),status varchar(10),foreign key (emp_id) references attandance_table(emp_id));
